def corr2d(X, K):
    h, w = K.shape
    Y = torch.zeros((X.shape[0] - h + 1, X.shape[1] - w + 1))
    for i in range(Y.shape[0]):
        for j in range(Y.shape[1]):
            Y[i, j] = (X[i:i + h, j:j + w] * K).sum()
    return Y

conv2d = nn.Conv2d(1, 1, kernel_size=(1, 2), bias=False)

# иҝҷдёӘдәҢз»ҙеҚ·з§ҜеұӮдҪҝз”Ёеӣӣз»ҙиҫ“е…Ҙе’Ңиҫ“еҮәж јејҸпјҲжү№йҮҸеӨ§е°ҸгҖҒйҖҡйҒ“гҖҒй«ҳеәҰгҖҒе®ҪеәҰпјүпјҢ
X = X.reshape((1, 1, 6, 8))
Y = Y.reshape((1, 1, 6, 7))

lr = 3e-2

for i in range(10):
    Y_hat = conv2d(X)
    l = (Y_hat - Y) ** 2
    conv2d.zero_grad()
    l.sum().backward()

    # иҝӯд»ЈеҚ·з§Ҝж ё
    conv2d.weight.data[:] -= lr * conv2d.weight.grad
    if (i + 1) % 2 == 0:
        print(f'epoch {i+1}, loss {l.sum():.3f}')
        
print(f'еҚ·з§Ҝж ёпјҡ{conv2d.weight.data.reshape((1, 2))}')

2.4 зү№еҫҒжҳ е°„е’Ңж„ҹеҸ—йҮҺ

еҚ·з§ҜеұӮеҸҜд»Ҙиў«и§ҶдёәдёҖдёӘиҫ“е…Ҙжҳ е°„еҲ°дёӢдёҖеұӮзҡ„з©әй—ҙз»ҙеәҰзҡ„иҪ¬жҚўеҷЁпјҢеӣ жӯӨд№ҹиў«з§°дёәзү№еҫҒжҳ е°„гҖӮ

еңЁеҚ·з§ҜзҘһз»ҸзҪ‘з»ңдёӯпјҢеҜ№дәҺжҹҗдёҖеұӮзҡ„д»»ж„Ҹе…ғзҙ xxxпјҢе…¶ж„ҹеҸ—йҮҺжҳҜжҢҮеңЁеүҚеҗ‘дј ж’ӯжңҹй—ҙеҸҜиғҪеҪұе“Қи®Ўз®—зҡ„жүҖжңүе…ғзҙ пјҲжқҘиҮӘжүҖжңүе…ҲеүҚеұӮпјүгҖӮ

иҜ·жіЁж„ҸпјҢж„ҹеҸ—йҮҺеҸҜиғҪеӨ§дәҺиҫ“е…Ҙзҡ„е®һйҷ…еӨ§е°ҸгҖӮд»ҘжҲ‘д»¬дёҠиҫ№дҪ“з§Ҝзҡ„ 2Г—2 еҚ·з§Ҝж ёдёәдҫӢпјҢйҳҙеҪұиҫ“еҮәе…ғзҙ еҖј 19 зҡ„ж„ҹеҸ—йҮҺжҳҜиҫ“е…ҘйҳҙеҪұйғЁеҲҶзҡ„еӣӣдёӘе…ғзҙ гҖӮеҒҮи®ҫд№ӢеүҚиҫ“еҮәдёә Y\boldsymbol{Y}YпјҢе…¶еӨ§е°Ҹдёә 2Г—2пјҢзҺ°еңЁжҲ‘д»¬еңЁе…¶еҗҺйҷ„еҠ дёҖдёӘеҚ·з§ҜеұӮпјҢиҜҘеҚ·з§ҜеұӮд»Ҙ Y\boldsymbol{Y}Y дёәиҫ“е…ҘпјҢиҫ“еҮәеҚ•дёӘе…ғзҙ  zzzгҖӮеңЁиҝҷз§Қжғ…еҶөдёӢпјҢY\boldsymbol{Y}Y дёҠзҡ„ zzz зҡ„ж„ҹеҸ—йҮҺеҢ…жӢ¬ Y\boldsymbol{Y}Y зҡ„жүҖжңүеӣӣдёӘе…ғзҙ пјҢиҖҢиҫ“е…Ҙзҡ„ж„ҹеҸ—йҮҺеҢ…жӢ¬жңҖеҲқжүҖжңүд№қдёӘиҫ“е…Ҙе…ғзҙ гҖӮ

еӣ жӯӨпјҢеҪ“дёҖдёӘзү№еҫҒеӣҫдёӯзҡ„д»»ж„Ҹе…ғзҙ йңҖиҰҒжЈҖжөӢжӣҙе№ҝеҢәеҹҹзҡ„иҫ“е…Ҙзү№еҫҒж—¶пјҢжҲ‘д»¬еҸҜд»Ҙжһ„е»әдёҖдёӘжӣҙж·ұзҡ„зҪ‘з»ңгҖӮ

2.5 еЎ«е……

еҚ·з§ҜеҗҺзҡ„еӣҫеғҸе°әеҜёзј©е°ҸдёҖдәӣгҖӮеҸҜд»ҘиҜҒжҳҺпјҢиӢҘиҫ“е…ҘеӣҫеғҸзҡ„еҪўзҠ¶дёә nhГ—nwn_h \times n_wnhвҖӢГ—nwвҖӢпјҢеҚ·з§Ҝж ёзҡ„еҪўзҠ¶дёә khГ—kwk_h \times k_wkhвҖӢГ—kwвҖӢпјҢеҲҷеҚ·з§ҜеҗҺеӣҫеғҸзҡ„е°әеҜёдёә

(nhвҲ’kh+1)Г—(nwвҲ’kw+1)(n_h - k_h + 1) \times (n_w - k_w + 1) (nhвҖӢвҲ’khвҖӢ+1)Г—(nwвҖӢвҲ’kwвҖӢ+1)

еӣ жӯӨеҪ“жҲ‘д»¬иҝӣиЎҢеӨҡеұӮеҚ·з§Ҝж—¶пјҢиҫ№зјҳеғҸзҙ дјҡдёҘйҮҚдёўеӨұгҖӮдёәдәҶзј“и§Јиҝҷз§Қжғ…еҶөпјҢжҲ‘д»¬еёёеёёеңЁеӣҫеғҸиҫ№зјҳеЎ«е……дёҖдәӣеҖјдёә 0 зҡ„еғҸзҙ гҖӮ

еҪ“й«ҳеәҰеЎ«е……е’Ңе®ҪеәҰеЎ«е……еҲҶеҲ«дёә php_hphвҖӢ е’Ң pwp_wpwвҖӢ ж—¶пјҢиҫ“еҮәеӣҫеғҸзҡ„е°әеҜёдёә

(nhвҲ’kh+ph+1)Г—(nwвҲ’kw+pw+1)(n_h - k_h + p_h + 1) \times (n_w - k_w + p_w + 1) (nhвҖӢвҲ’khвҖӢ+phвҖӢ+1)Г—(nwвҖӢвҲ’kwвҖӢ+pwвҖӢ+1)

еӣ жӯӨпјҢеҪ“жҲ‘д»¬йңҖиҰҒеҫ—еҲ°е’Ңиҫ“е…Ҙе°әеҜёзӣёеҗҢзҡ„иҫ“еҮәж—¶пјҢеҸӘйңҖиҰҒи®ҫзҪ® php_hphвҖӢ е’Ң pwp_wpwвҖӢ еҲҶеҲ«дёә khвҲ’1k_h - 1khвҖӢвҲ’1 е’Ң kwвҲ’1k_w - 1kwвҖӢвҲ’1 еҚіеҸҜгҖӮ

PyTorch жҸҗдҫӣдәҶеЎ«е……зҡ„е®һзҺ°пјҡ

nn.Conv2d(1, 1, kernel_size=3, padding=1) # еңЁжүҖжңүдҫ§иҫ№еЎ«е…… 1 дёӘеғҸзҙ 
nn.Conv2d(1, 1, kernel_size=3, padding=(2, 1)) # еңЁй«ҳеәҰдёҠеЎ«е…… 2 дёӘеғҸзҙ пјҢеңЁе®ҪеәҰдёҠеЎ«е…… 1 дёӘеғҸзҙ 

2.6 жӯҘе№…

еҚ·з§Ҝж ёжҜҸж¬Ўж»‘еҠЁзҡ„и·қзҰ»з§°дёәжӯҘе№…гҖӮжҲ‘д»¬д№ӢеүҚдҪҝз”Ёзҡ„йғҪжҳҜй•ҝеәҰдёә 1 зҡ„жӯҘе№…гҖӮжңүж—¶дёәдәҶй«ҳж•Ҳи®Ўз®—пјҢжҲ–иҖ…дёәдәҶйҷҚдҪҺйҮҮж ·ж¬Ўж•°пјҢжҲ‘д»¬еҸҜд»ҘдҪҝеҚ·з§Ҝж ёи·іиҝҮйғЁеҲҶеғҸзҙ пјҢжҜҸж¬Ўж»‘еҠЁеӨҡдёӘеғҸзҙ гҖӮ

дҫӢеҰӮиҝҷжҳҜж°ҙе№іжӯҘе№…дёә 2пјҢеһӮзӣҙжӯҘе№…дёә 3 зҡ„з»“жһңпјҡ

е®№жҳ“и®Ўз®—пјҢеҪ“еһӮзӣҙжӯҘе№…е’Ңж°ҙе№іжӯҘе№…еҲҶеҲ«дёә shs_hshвҖӢ е’Ң sws_wswвҖӢ ж—¶пјҢиҫ“еҮәеӣҫеғҸзҡ„е°әеҜёдёә

вҢҠnhвҲ’kh+ph+shshвҢӢГ—вҢҠnwвҲ’kw+pw+swswвҢӢ\lfloor \frac{n_h - k_h + p_h + s_h}{s_h} \rfloor \times \lfloor \frac{n_w - k_w + p_w + s_w}{s_w} \rfloor вҢҠshвҖӢnhвҖӢвҲ’khвҖӢ+phвҖӢ+shвҖӢвҖӢвҢӢГ—вҢҠswвҖӢnwвҖӢвҲ’kwвҖӢ+pwвҖӢ+swвҖӢвҖӢвҢӢ

PyTorch еҗҢж ·жҸҗдҫӣдәҶжӯҘе№…зҡ„е®һзҺ°пјҡ

nn.Conv2d(1, 1, kernel_size=3, padding=1, stride=2) # жӯҘе№…дёә 2
nn.Conv2d(1, 1, kernel_size=3, padding=1, stride=(3, 4)) # еһӮзӣҙжӯҘе№…дёә 3пјҢж°ҙе№іжӯҘе№…дёә 4

2.7 йҖҡйҒ“

2.7.1 еӨҡиҫ“е…ҘйҖҡйҒ“

еҪ“жҲ‘д»¬ж·»еҠ йҖҡйҒ“ж—¶пјҢиҫ“е…ҘеӣҫеғҸе°ұеҸҳжҲҗдәҶдёҖдёӘдёүз»ҙеј йҮҸпјҢе®ғзҡ„еӨ§е°Ҹдёә

hГ—wГ—ch \times w \times c hГ—wГ—c

еҜ№дәҺжңҖеёёи§Ғзҡ„ RGB еӣҫеғҸпјҢccc дёә 3гҖӮ

еҪ“иҫ“е…ҘеҢ…еҗ«еӨҡдёӘйҖҡйҒ“ж—¶пјҢйңҖиҰҒжһ„йҖ дёҖдёӘзӣёеҗҢйҖҡйҒ“ж•°йҮҸзҡ„еҚ·з§Ҝж ёпјҢд»ҘдҫҝеҜ№еӨҡдёӘйҖҡйҒ“иҝӣиЎҢдә’зӣёе…іиҝҗз®—гҖӮеӣ жӯӨеҜ№дәҺ cic_iciвҖӢ йҖҡйҒ“еӣҫеғҸпјҢе…¶еҚ·з§Ҝж ёеӨ§е°Ҹдёә

khГ—kwГ—cik_h \times k_w \times c_i khвҖӢГ—kwвҖӢГ—ciвҖӢ

дҫӢеҰӮдёҖдёӘеҸҢйҖҡйҒ“еӣҫеғҸзҡ„еҸҢйҖҡйҒ“еҚ·з§Ҝж ёпјҡ

2.7.2 еӨҡиҫ“еҮәйҖҡйҒ“

жңүж—¶жҲ‘д»¬еёҢжңӣеҫ—еҲ°еӨҡдёӘиҫ“еҮәйҖҡйҒ“пјҢжҜҸдёӘйҖҡйҒ“д»ЈиЎЁдёҚеҗҢзҡ„иҫ“еҮәе“Қеә”пјҢеҰӮиҫ№зјҳгҖҒзә№зҗҶгҖҒеҪўзҠ¶зӯүгҖӮжҜҸж¬ЎеҚ·з§ҜйғҪжҳҜеңЁйҷҚдҪҺиҫ“е…ҘеӣҫеғҸзҡ„еӨ§е°ҸпјҢеҗҢж—¶жҲ‘д»¬еҸҜд»ҘеўһеҠ иҫ“еҮәйҖҡйҒ“зҡ„ж•°йҮҸпјҢд»ҘжҚ•жҚүжӣҙеӨҡжӣҙеӨҚжқӮзҡ„жҠҪиұЎзү№еҫҒгҖӮ

дёәдәҶиҺ·еҫ—еӨҡдёӘйҖҡйҒ“зҡ„иҫ“еҮәпјҢжҲ‘д»¬еҸҜд»ҘдёәжҜҸдёӘиҫ“еҮәйҖҡйҒ“еҲӣе»әдёҖдёӘеҚ·з§Ҝж ёгҖӮжҜҸдёӘеҚ·з§Ҝж ёеҜ№еҺҹе§ӢеӣҫеғҸиҝӣиЎҢи®Ўз®—еҫ—еҲ°еҜ№еә”йҖҡйҒ“зҡ„иҫ“еҮәгҖӮ

иҝҷж ·еҚ·з§Ҝж ёзҡ„еӨ§е°Ҹдёә

khГ—kwГ—ciГ—cok_h \times k_w \times c_i \times c_o khвҖӢГ—kwвҖӢГ—ciвҖӢГ—coвҖӢ

4 жұ еҢ–еұӮ

жҲ‘д»¬еңЁдёҠиҫ№жҸҗеҲ°дәҶж„ҹеҸ—йҮҺгҖӮеңЁеҚ·з§ҜзҘһз»ҸзҪ‘з»ңдёӯпјҢжҲ‘д»¬еёҢжңӣйҡҸзқҖзҪ‘з»ңеұӮж•°еҠ ж·ұпјҢйҖҗжӯҘйҷҚдҪҺзү№еҫҒеӣҫзҡ„з©әй—ҙеҲҶиҫЁзҺҮпјҢд»ҺиҖҢиҒҡеҗҲдҝЎжҒҜгҖҒжү©еӨ§ж„ҹеҸ—йҮҺпјҢи®©зҘһз»Ҹе…ғеҜ№жӣҙеӨ§зҡ„еӣҫеғҸеҢәеҹҹдҪңеҮәе“Қеә”гҖӮиҝҷз§ҚеҒҡжі•еҸҜд»Ҙи®©зҪ‘з»ңжңҖз»Ҳе…·еӨҮеҜ№ж•ҙеј еӣҫеғҸиҝӣиЎҢе…ЁеұҖеҲӨж–ӯзҡ„иғҪеҠӣпјҢ

з„¶иҖҢпјҢзҺ°е®һдёӯеӣҫеғҸеҫҖеҫҖеӯҳеңЁдҪҚзҪ®еҒҸз§»пјҲеҰӮжӢҚж‘„и§’еәҰгҖҒжҠ–еҠЁзӯүпјүпјҢжҲ‘д»¬дёҚеёҢжңӣжЁЎеһӢеҜ№иҝҷдәӣеҫ®е°ҸеҸҳеҢ–иҝҮдәҺж•Ҹж„ҹгҖӮеӣ жӯӨпјҢжҲ‘д»¬йҷӨдәҶдҪҝз”ЁеҚ·з§Ҝж“ҚдҪңпјҢиҝҳдјҡеј•е…Ҙжұ еҢ–еұӮпјҲд№ҹз§°жұҮиҒҡеұӮпјүжқҘиҝӣдёҖжӯҘйҷҚдҪҺз©әй—ҙеҲҶиҫЁзҺҮпјҢеҠ еҝ«жЁЎеһӢи®Ўз®—пјҢжҸҗеҸ–жӣҙзЁіе®ҡзҡ„зү№еҫҒпјҲеҰӮиҫ№зјҳгҖҒи§’зӮ№пјүпјҢе№¶еўһејәзҪ‘з»ңзҡ„е№із§»дёҚеҸҳжҖ§пјҢйҷҚдҪҺеҜ№еӣҫеғҸдёӯзү©дҪ“е…·дҪ“дҪҚзҪ®зҡ„ж•Ҹж„ҹжҖ§гҖӮ

4.1 жңҖеӨ§жұ еҢ–дёҺе№іеқҮжұ еҢ–

дёҺеҚ·з§ҜеұӮзұ»дјјпјҢжұ еҢ–иҝҗз®—з¬Ұд№ҹжҳҜдёҖдёӘеӣәе®ҡеҪўзҠ¶зҡ„зӘ—еҸЈгҖӮдёҺеҚ·з§Ҝж ёдёҚеҗҢзҡ„жҳҜпјҢжұ еҢ–иҝҗз®—жІЎжңүеҸӮж•°пјҢиҖҢжҳҜи®Ўз®—зӘ—еҸЈдёӯжүҖжңүе…ғзҙ зҡ„жңҖеӨ§еҖјжҲ–е№іеқҮеҖјпјҢеҲҶеҲ«з§°дёәжңҖеӨ§жұ еҢ–е’Ңе№іеқҮжұ еҢ–гҖӮ

дҫӢеҰӮжңҖеӨ§жұ еҢ–пјҡ

PyTorch жҸҗдҫӣдәҶжңҖеӨ§жұ еҢ–еұӮе’Ңе№іеқҮжұ еҢ–еұӮпјҢеҸҜд»ҘзӣҙжҺҘдҪҝз”ЁгҖӮ

4.2 еЎ«е……е’ҢжӯҘе№…

дёҺеҚ·з§ҜеұӮдёҖж ·пјҢжұ еҢ–еұӮд№ҹеҸҜд»ҘйҖҡиҝҮеЎ«е……е’ҢжӯҘе№…ж”№еҸҳиҫ“еҮәеҪўзҠ¶гҖӮй»ҳи®Өжғ…еҶөдёӢпјҢPyTorch дёӯзҡ„жӯҘе№…е’Ңжұ еҢ–зӘ—еҸЈзҡ„еӨ§е°ҸзӣёеҗҢгҖӮ

nn.MaxPool2d(3) # жңҖеӨ§жұ еҢ–пјҢжұ еҢ–зӘ—еҸЈдёә 3пјҢжӯҘе№…дёә 3
nn.MaxPool2d(3, padding=1, stride=2) # жңҖеӨ§жұ еҢ–пјҢжұ еҢ–зӘ—еҸЈдёә 3пјҢеЎ«е……дёә 1пјҢжӯҘе№…дёә 2
nn.AvgPool2d((2, 3), padding=(0, 1), stride=(2, 3)) # е№іеқҮжұ еҢ–пјҢжұ еҢ–зӘ—еҸЈдёә 2Г—3пјҢй«ҳеәҰеЎ«е…… 0пјҢе®ҪеәҰдёӨдҫ§еЎ«е…… 1пјҢеһӮзӣҙжӯҘе№…дёә 2пјҢж°ҙе№іжӯҘе№…дёә 3

4.3 йҖҡйҒ“

еңЁеӨ„зҗҶеӨҡйҖҡйҒ“ж•°жҚ®ж—¶пјҢжұ еҢ–еұӮеңЁжҜҸдёӘиҫ“е…ҘйҖҡйҒ“дёҠеҚ•зӢ¬иҝҗз®—пјҢе№¶дёҚдјҡеңЁиҫ“еҮәйҖҡйҒ“дёӯжұҮжҖ»гҖӮеӣ жӯӨжұ еҢ–еұӮзҡ„иҫ“еҮәйҖҡйҒ“е’Ңиҫ“е…ҘйҖҡйҒ“ж•°йҮҸзӣёеҗҢгҖӮ
